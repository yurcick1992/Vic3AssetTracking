# State Assets Visualization v1.3 — Scripted Effects
# Five groups of metrics with yearly snapshots (up to 5 years) and deltas:
#   1. Government-owned domestic building levels + privatization value (Assets tab)
#   2. Government net worth = treasury + gov asset value (Assets tab)
#   3. National industry levels: domestic + foreign, split columns (Capital tab)
#   4. National industry value: domestic + foreign, split columns (Capital tab)
#   5. National capital: dom_value + for_value + treasury + ipool (Capital tab)

# ─── COUNTING (Government-owned domestic) ───
# Count government-owned building levels and privatization value in ROOT's states,
# then capture treasury and compute net worth.
# Must be called in country scope.
inv_count_gov_building_levels_effect = {
	# Reset counters
	set_variable = { name = gov_levels_current value = 0 }
	set_variable = { name = gov_value_current value = 0 }

	# Sum across all privatizable buildings
	every_scope_building = {
		limit = {
			is_government_funded = no
			is_subsistence_building = no
			NOT = { is_building_group = bg_service }
			NOT = { is_building_group = bg_owner_buildings }
			NOT = { is_building_group = bg_gold_fields }
			NOT = { is_building_group = bg_monuments_hidden }
			country_ownership_fraction > 0
		}
		ROOT = {
			change_variable = { name = gov_levels_current add = PREV.inv_gov_owned_levels_value }
			change_variable = { name = gov_value_current add = PREV.inv_gov_owned_value_per_building }
		}
	}

	# Capture treasury: cash reserves minus debt principal
	set_variable = { name = treasury_current value = gold_reserves }
	change_variable = { name = treasury_current subtract = principal }

	# Compute net worth = treasury + state asset value
	set_variable = { name = total_current value = var:treasury_current }
	change_variable = { name = total_current add = var:gov_value_current }
}

# ─── COUNTING (National assets: domestic + foreign, split) ───
# Count ALL nationally-owned building levels and value (gov + private + self combined)
# across both domestic states and foreign countries where ROOT has investment.
# Uses the staircase approach (levels_owned_by_country) for accurate counting.
# Also captures investment pool and computes national capital sum.
# Must be called in country scope AFTER inv_count_gov_building_levels_effect
# (requires treasury_current to be set).
inv_count_national_assets_effect = {
	# Reset counters — domestic and foreign tracked separately
	set_variable = { name = nat_dom_levels_current value = 0 }
	set_variable = { name = nat_dom_value_current value = 0 }
	set_variable = { name = nat_for_levels_current value = 0 }
	set_variable = { name = nat_for_value_current value = 0 }

	# Save ROOT for gdp_ownership_ratio filter in nested scopes
	save_scope_as = inv_root

	# ── Domestic buildings ──
	# levels_owned_by_country(ROOT) counts all of ROOT's nationals' ownership
	# (government + private + self combined).
	every_scope_building = {
		limit = {
			is_government_funded = no
			is_subsistence_building = no
			NOT = { is_building_group = bg_service }
			NOT = { is_building_group = bg_owner_buildings }
			NOT = { is_building_group = bg_gold_fields }
			NOT = { is_building_group = bg_monuments_hidden }
			levels_owned_by_country = { target = ROOT value >= 1 }
		}
		inv_staircase_nat_levels_effect = yes
		inv_compute_nat_value_for_building_effect = yes
		ROOT = {
			change_variable = { name = nat_dom_levels_current add = var:temp_bldg_nat_levels }
			change_variable = { name = nat_dom_value_current add = var:temp_bldg_nat_value }
		}
	}

	# ── Foreign buildings ──
	# Pre-filter: only iterate countries where ROOT has any investment (O(countries) check).
	every_country = {
		limit = {
			NOT = { this = scope:inv_root }
			"gdp_ownership_ratio(scope:inv_root)" > 0
		}
		every_scope_building = {
			limit = {
				is_government_funded = no
				is_subsistence_building = no
				NOT = { is_building_group = bg_service }
				NOT = { is_building_group = bg_owner_buildings }
				NOT = { is_building_group = bg_gold_fields }
				NOT = { is_building_group = bg_monuments_hidden }
				levels_owned_by_country = { target = ROOT value >= 1 }
			}
			inv_staircase_nat_levels_effect = yes
			inv_compute_nat_value_for_building_effect = yes
			scope:inv_root = {
				change_variable = { name = nat_for_levels_current add = var:temp_bldg_nat_levels }
				change_variable = { name = nat_for_value_current add = var:temp_bldg_nat_value }
			}
		}
	}

	# ── Sum columns (derived) ──
	set_variable = { name = nat_sum_levels_current value = var:nat_dom_levels_current }
	change_variable = { name = nat_sum_levels_current add = var:nat_for_levels_current }

	set_variable = { name = nat_sum_value_current value = var:nat_dom_value_current }
	change_variable = { name = nat_sum_value_current add = var:nat_for_value_current }

	# Capture investment pool
	set_variable = { name = ipool_current value = investment_pool }

	# Compute national capital = sum_value + treasury + ipool
	set_variable = { name = natcap_current value = var:nat_sum_value_current }
	change_variable = { name = natcap_current add = var:treasury_current }
	change_variable = { name = natcap_current add = var:ipool_current }
}
